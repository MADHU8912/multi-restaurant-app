${r.name}

`; grid.appendChild(card); }); } function loadMenu(id, name) { document.getElementById("menuTitle").innerText = name + " Menu"; const menuList = document.getElementById("menuList"); menuList.innerHTML = ""; const items = menuItems.filter(i => i.restaurant_id === id); items.forEach(item => { const div = document.createElement("div"); div.className = "menu-item"; div.innerHTML = `

${item.name}

₹${item.price}

`; menuList.appendChild(div); }); } loadRestaurants();
